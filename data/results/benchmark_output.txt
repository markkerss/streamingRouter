class Snake:
    def __init__(self):
        """
        Initialize snake object
        """
        # setup snake object
        self.body = [(1, 1), (1, 2), (1, 3)]
        self.direction = (1, 0)
    
    def update(self):
        """
        Update snake position
        """
        # check for collisions with walls
        for wall in WALL:
            if is_collision(self.body[0], wall):
                return None
        
        # update snake position
        self.body.insert(0, self.direction)
        
        # check for collisions with self
        if is_collision(self.body[-1], self.body[0]):
            return None
        
        # return updated snake object
        return self
    
    def move(self):
        """
        Move snake in direction
        """
        # update snake position
        new_body = self.update()
        if new_body is None:
            return None
        
        # check for food
        if is_food(new_body[0]):
            return self.eat_food()
        
        # return new snake object
        return new_body
    
    def eat_food(self):
        """
        Move food to snake
        """
        # create new head position
        head = self.body[0]
        self.body.insert(0, (self.body[1][0], self.body[1][1]))
        
        # set new head
        self.body[0] = head
        
        # check for collisions with walls
        for wall in WALL:
            if is_collision(head, wall):
                return None
        
        # return updated snake object
        return self
